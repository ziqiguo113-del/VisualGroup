#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip> 
#include <string>
#include <vector>
using namespace std;

int f(int day, int n) {
    if (day == n) {
        return 1;  
    }
    return (f(day + 1, n) + 1) * 2;
}

int main() {
    int n;
    cin >> n;

    cout << f(1, n) << endl; 

    return 0;
}
