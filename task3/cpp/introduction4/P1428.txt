#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip> 
using namespace std;
int main() {
    int n;
    cin >> n;
    int q[100];
    for (int i = 0; i <n;i++) {
        cin >> q[i];
    }
    for (int i = 0; i < n; i++) {
        int count = 0;
        for (int j = 0; j < i; j++) {
            if (q[j] < q[i]) {
                count++;
            }
        }
        cout << count << " ";
    }
    return 0;
}
