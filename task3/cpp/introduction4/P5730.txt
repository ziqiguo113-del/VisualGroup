#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip> 
#include <string>
using namespace std;

const string digits[10][5] = {
    {"###", "# #", "# #", "# #", "###"},  
    {"  #", "  #", "  #", "  #", "  #"}, 
    {"###", "  #", "###", "#  ", "###"}, 
    {"###", "  #", "###", "  #", "###"}, 
    {"# #", "# #", "###", "  #", "  #"}, 
    {"###", "#  ", "###", "  #", "###"}, 
    {"###", "#  ", "###", "# #", "###"}, 
    {"###", "  #", "  #", "  #", "  #"}, 
    {"###", "# #", "###", "# #", "###"}, 
    {"###", "# #", "###", "  #", "###"}  
};

int main() {
    int n;
    string s;
    cin >> n >> s;

    for (int row = 0; row < 5; row++) {
        for (int i = 0; i < n; i++) {
            int digit = s[i] - '0';  
            cout << digits[digit][row];
            if (i < n - 1) cout << " ";  
        }
        cout << endl;
    }

    return 0;
}
