#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip> 
using namespace std;
int main() {
    int n;
    cin >> n;
    bool win[34] = { false };
    for (int i = 0; i < 7; i++) {
        int num;
        cin >> num;
        win[num] = true;
    }
    int prize[7] = { 0 };
    for (int i = 0; i < n; i++) {
        int count = 0;
        for (int j = 0; j < 7; j++) {
            int num;
            cin >> num;
            if (win[num])count++;
        }
        if (count > 0)prize[7 - count]++;
    }
    for (int i=0;i<7;i++){
        cout << prize[i] << " ";
    }
    return 0;
}
