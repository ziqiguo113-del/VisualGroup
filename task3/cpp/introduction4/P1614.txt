#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip> 
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    int a[3000];  

    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int min_sum = 1e9;  
    for (int i = 0; i <= n - m; i++) {
        int sum = 0;
        for (int j = i; j < i + m; j++) {
            sum += a[j];
        }
        if (sum < min_sum) {
            min_sum = sum;
        }
    }

    cout << min_sum << endl;
    return 0;
}
