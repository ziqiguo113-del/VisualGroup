#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip> 
using namespace std;
int main(){
    int n;
    cin >> n;
    double sqrt5 = sqrt(5);
    double num = (pow((1 + sqrt5) / 2, n) - pow((1 - sqrt5) / 2, n)) / sqrt5;
    cout << fixed << setprecision(2) << num << endl;
    return 0;
}
