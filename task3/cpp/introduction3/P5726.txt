#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip> 
using namespace std;
int main() {
    int n;
    cin >> n;
    int scores[1000];
    for (int i = 0; i < n; i++) {
        cin >> scores[i];
    }
    int max_score = scores[0], min_score = scores[0];
    for (int i = 1; i < n; i++) {
        if (scores[i] > max_score) max_score = scores[i];
        if (scores[i] < min_score) min_score = scores[i];
    }
    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += scores[i];
    }
    sum = sum - max_score - min_score;
    double average = (double)sum / (n - 2);
    cout << fixed << setprecision(2) << average << endl;
    return 0;
}
