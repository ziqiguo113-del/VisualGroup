#include <iostream>
#include <algorithm>
using namespace std;
int main(){
    int n;
    cin >> n;
    int min = 1001;
    for (int i = 0; i < n; i++) {
        int num;
        cin >> num;
        if(num<min){
            min = num;
        }
    }
    cout << min << endl;
    return 0;
}
