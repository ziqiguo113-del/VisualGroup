#include <iostream>
#include <algorithm>
using namespace std;
int main(){
    int k;
    cin >> k;
    int days_left=k;
    int current_group=1;
    int num=0;
    while (days_left > 0) {
        int days_to_pay = min(days_left , current_group);
        num += current_group * days_to_pay;
        days_left -= days_to_pay;
        current_group++;
    }
    cout << num << endl;
    return 0;
}
